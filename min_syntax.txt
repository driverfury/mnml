VAR = 'var'
FUNC = 'func'

VOID = 'void'
CHAR = 'char'
INT = 'int'
STRUCT = 'struct'

RETURN = 'return'
IF = 'if'
ELSE = 'else'
WHILE = 'while'

CAST = 'cast'

SPMEMB = '->'

IDENT = [_A-Za-z][_A-Za-z0-9]*
INTLIT = [0-9]*

<unit> = <glob_decl>*

<glob_decl> = VAR IDENT ':' <typespec> ';'
            | FUNC IDENT '(' <params>? ')' ':' <typespec> '{' (<decl> ',')* <stmt>* '}'
            | STRUCT IDENT <struct_def>

<params> = <param> (','<param>))*

<stmt_block> = '{' <stmt>* '}'

<stmt> = IF '(' <expr> ')' <stmt> (ELSE <stmt>)?
       | WHILE '(' <expr> ')' <stmt>
       | RETURN <expr>? ';'
       | '{' <stmt>* '}'
       | IDENT '(' ')' ';'
       | <lvalue> '=' <expr> ';'

<lvalue> = IDENT
         | IDENT '[' <expr> ']'
         | IDENT '.' IDENT
         | IDENT SPMEMB IDENT
         | '*' <lvalue>
         | '(' <lvalue> ')'

<decl> = VAR IDENT ':' <typespec>

<typespec> = VOID
           | CHAR
           | INT
           | <typeref>
           | '[' typespec ']'
           | <typespec> '[' <expr> ']'

<struct_def> = '{' (IDENT ':' <typespec> ';')+ '}'

<args> = <expr> (',' <expr>)*

<expr_base> = INTLIT
            | IDENT
            | '(' <expr> ')'

<expr_first> = <expr_base> '(' <args>? ')'
             | <expr_base> '[' <expr> ']'
             | <expr_base> '.' IDENT
             | <expr_base> SPMEMB IDENT
             | <expr_base>

<expr_unary> = '+' <expr_unary>
             | '-' <expr_unary>
             | '*' <expr_unary>
             | '&' <expr_unary>
             | CAST '<' <typespec> '>' <expr_first>
             | <expr_first>

<expr_mul> = <expr_mul> '*' <expr_unary>
           | <expr_mul> '/' <expr_unary>
           | <expr_mul> '%' <expr_unary>
           | <expr_unary>

<expr_add> = <expr_add> '+' <expr_mul>
           | <expr_add> '-' <expr_mul>
           | <expr_mul>

<expr> = <expr_add>
